    // Format file size
    function formatFileSize(bytes) {
      if (bytes === 0) return '0 Bytes';
      const k = 1024;
      const sizes = ['Bytes', 'KB', 'MB', 'GB'];
      const i = Math.floor(Math.log(bytes) / Math.log(k));
      return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
    }

    async function loadPdf(pdf) {

          <div style="display: flex; flex-direction: column; align-items: center; justify-content: center; height: 100%;">

    // Render PDF page
    async function renderPage(num) {
      try {
        const page = await pdfDoc.getPage(num);
        
        // Calculate scale to fit width
        const viewport = page.getViewport({ scale: 1 });
        const containerWidth = pdfViewerContainer.clientWidth - 40; // Account for padding
        const initialScale = containerWidth / viewport.width;
        
        // Skapa eller återanvänd canvas
        let canvas = document.getElementById('pdf-canvas');
        if (!canvas) {
          // Reset the container
          pdfViewerContainer.innerHTML = '';
          
          // Skapa canvas
          canvas = document.createElement('canvas');
          canvas.id = 'pdf-canvas';
          pdfViewerContainer.appendChild(canvas);
          
          // Återställ scale när en ny PDF visas
          scale = initialScale;
          
          // Add zoom and pan functionality
          setupZoomAndPan(pdfViewerContainer, canvas);
          
          // Uppdatera zoom-nivå i det gröna fältet när PDF:en öppnas
          const zoomLevelElement = document.getElementById('zoom-level');
          if (zoomLevelElement) {
            zoomLevelElement.textContent = `${Math.round(scale * 100)}%`;
          }
        }
        
        // Använd aktuell scale för viewport
        const scaledViewport = page.getViewport({ scale: initialScale });
        
        const context = canvas.getContext('2d');
        canvas.height = scaledViewport.height;
        canvas.width = scaledViewport.width;
        
        // Återställ transform för att undvika dubbel-scaling
        canvas.style.transform = `scale(${scale})`;
        canvas.style.transformOrigin = '0 0';
        
        const renderContext = {
          canvasContext: context,
          viewport: scaledViewport
        };
        
        await page.render(renderContext).promise;
        
        // Extract text content for speech synthesis
        try {
          const textContent = await page.getTextContent();
          const textItems = textContent.items;
          
          // Create a string with proper spacing based on the text positions
          let lastY;
          let text = '';
          
          for (let i = 0; i < textItems.length; i++) {
            const item = textItems[i];
            
            // Check if we're on a new line by comparing y-coordinates
            if (lastY !== undefined && Math.abs(lastY - item.transform[5]) > 5) {
              text += '\n'; // Add newline for significant y-position changes
            } else if (i > 0 && item.hasEOL) {
              text += '\n';
            }
            
            text += item.str;
            lastY = item.transform[5];
          }
          
          // Store the extracted text for this page
          extractedTextContent[num] = text;
          
        } catch (error) {
          console.error('Error extracting text content:', error);
          extractedTextContent[num] = 'Text could not be extracted from this page.';
        }
        
        // Center canvas horizontally if needed
        if (canvas.width < pdfViewerContainer.clientWidth) {
          const margin = Math.max(0, (pdfViewerContainer.clientWidth - canvas.width) / 2);
          canvas.style.marginLeft = margin + 'px';
        } else {
          canvas.style.marginLeft = '0';
        }
        
        // Update page info with zoom level
        updatePageInfo();
      } catch (error) {
        console.error('Error rendering page:', error);
        pdfViewerContainer.innerHTML = `
          <div style="text-align: center; color: white; padding: 2rem;">
            <h3>Kunde inte visa PDF</h3>
            <p>${error.message || 'Ett fel uppstod vid rendering av PDF.'}</p>
            <p>Kontrollera att PDF-filen existerar på servern.</p>
          </div>
        `;
      }
    }

    // Setup zoom and pan with improved canvas handling
    function setupZoomAndPan(container, canvas) {
      // Skapa en wrapper för canvasen som låter den expander korrekt vid zoom
      let canvasWrapper = document.getElementById('canvas-wrapper');
      if (!canvasWrapper) {
        canvasWrapper = document.createElement('div');
        canvasWrapper.id = 'canvas-wrapper';
        canvasWrapper.style.position = 'relative';
        canvasWrapper.style.display = 'inline-block';
        canvasWrapper.style.padding = '50px'; // Mycket extra utrymme runt kanvas
        canvasWrapper.style.transformOrigin = '0 0';
        canvasWrapper.style.margin = '50px'; // Extra marginal runt hela wrappern
        canvas.parentNode.insertBefore(canvasWrapper, canvas);
        canvasWrapper.appendChild(canvas);
      }
      
      container.addEventListener('wheel', function(e) {
        if (e.ctrlKey) {
          e.preventDefault();
          
          const oldScale = scale;
          
          // Get current mouse position relative to canvas
          const rect = canvas.getBoundingClientRect();
          const mouseX = e.clientX - rect.left;
          const mouseY = e.clientY - rect.top;
          
          // Get scroll position before zoom
          const scrollXBefore = container.scrollLeft;
          const scrollYBefore = container.scrollTop;
          
          // Calculate position on canvas before zoom
          const canvasXBefore = scrollXBefore + mouseX;
          const canvasYBefore = scrollYBefore + mouseY;
          
          // Change scale based on scroll direction
          if (e.deltaY < 0) {
            scale = Math.min(5, scale * 1.1); // Zoom in with higher max zoom
          } else {
            scale = Math.max(0.5, scale / 1.1); // Zoom out
          }
          
          // Update canvas scale with wrapper
          canvas.style.transformOrigin = '0 0';
          canvas.style.transform = `scale(${scale})`;
          
          // Set much larger explicit dimensions på wrapper för att se till att det finns gott om plats även vid hög zoom
          const extraMargin = 500; // Mycket stor extra marginal för att undvika avklippning
          canvasWrapper.style.width = `${canvas.width * scale + extraMargin}px`;
          canvasWrapper.style.height = `${canvas.height * scale + extraMargin}px`;
          
          // Säkerställ att container har tillräckligt utrymme för att rulla
          container.style.minWidth = '100%';
          container.style.minHeight = '100%';
          
          // Se till att det finns tillräcklig marginal i botten vid hög zoom
          document.querySelector('.pdf-viewer-container').style.paddingBottom = `${extraMargin}px`;
          
          // Calculate position on canvas after zoom
          const canvasXAfter = canvasXBefore * (scale / oldScale);
          const canvasYAfter = canvasYBefore * (scale / oldScale);
          
          // Calculate new scroll position to keep mouse position fixed
          const scrollXAfter = canvasXAfter - mouseX;
          const scrollYAfter = canvasYAfter - mouseY;
          
          // Set new scroll position
          container.scrollLeft = scrollXAfter;
          container.scrollTop = scrollYAfter;
          
          // Update zoom level in the green zoom control area only
          const zoomLevelElement = document.getElementById('zoom-level');
          if (zoomLevelElement) {
            zoomLevelElement.textContent = `${Math.round(scale * 100)}%`;
          }
          
          // Update page info without the zoom percentage
          pageInfo.textContent = `Sida ${pageNum} av ${pdfDoc.numPages}`;
        }
      });
      
      // Set draggable cursor and initial state
      container.style.cursor = 'grab';
      
      // Pan with mouse drag - significantly improved
      container.addEventListener('mousedown', function(e) {
        // Allow dragging anywhere in the container, not just on canvas
        isDragging = true;
        startX = e.clientX;
        startY = e.clientY;
        container.style.cursor = 'grabbing';
        e.preventDefault(); // Prevent text selection during drag
      });
      
      container.addEventListener('mousemove', function(e) {
        if (isDragging) {
          const dx = e.clientX - startX;
          const dy = e.clientY - startY;
          
          container.scrollLeft -= dx;
          container.scrollTop -= dy;
          
          startX = e.clientX;
          startY = e.clientY;
        }
      });
      
      // Mouse up event - works even if mouse is released outside the container
      window.addEventListener('mouseup', function() {
        if (isDragging) {
          isDragging = false;
          container.style.cursor = 'grab';
        }
      });
      
      // Touch support for mobile devices
      container.addEventListener('touchstart', function(e) {
        if (e.touches.length === 1) {
          isDragging = true;
          startX = e.touches[0].clientX;
          startY = e.touches[0].clientY;
          e.preventDefault();
        }
      });
      
      container.addEventListener('touchmove', function(e) {
        if (isDragging && e.touches.length === 1) {
          const dx = e.touches[0].clientX - startX;
          const dy = e.touches[0].clientY - startY;
          
          container.scrollLeft -= dx;
          container.scrollTop -= dy;
          
          startX = e.touches[0].clientX;
          startY = e.touches[0].clientY;
          e.preventDefault();
        }
      });
      
      container.addEventListener('touchend', function() {
        isDragging = false;
      });
    }
    
    // Text-to-speech functions
    function startSpeechSynthesis() {
      if (!pdfDoc || !extractedTextContent[pageNum]) {
        showNotification('Ingen text att läsa upp från den aktuella sidan.');
        return;
      }
      
      if (isSpeaking) {
        stopSpeechSynthesis();
        return;
      }
      
      // Create new utterance
      speechUtterance = new SpeechSynthesisUtterance(extractedTextContent[pageNum]);
      
      // Set language to Swedish
      speechUtterance.lang = 'sv-SE';
      
      // Set speech rate (1 = normal speed)
      speechUtterance.rate = 1.0;
      
      // Store current page
      currentSpeechPage = pageNum;
      
      // Show visual indication that reading is active
      document.getElementById('read-aloud-btn').style.display = 'none';
      document.getElementById('stop-read-aloud-btn').style.display = 'inline-flex';
      
      // Create a highlight element to show text is being read
      const canvas = document.getElementById('pdf-canvas');
      if (canvas) {
        highlightedTextElement = document.createElement('div');
        highlightedTextElement.className = 'tts-highlight';
        highlightedTextElement.style.position = 'absolute';
        highlightedTextElement.style.top = '0';
        highlightedTextElement.style.left = '0';
        highlightedTextElement.style.width = '100%';
        highlightedTextElement.style.height = '100%';
        highlightedTextElement.style.pointerEvents = 'none';
        highlightedTextElement.style.boxShadow = 'inset 0 0 0 3px rgba(99, 102, 241, 0.5)';
        highlightedTextElement.style.zIndex = '10';
        canvas.parentNode.appendChild(highlightedTextElement);
      }
      
      // Handle end of speech
      speechUtterance.onend = function() {
        // If there are more pages and auto-continue is enabled, go to next page
        if (currentSpeechPage < pdfDoc.numPages) {
          pageNum = currentSpeechPage + 1;
          renderPage(pageNum).then(() => {
            updatePageInfo();
            // Short delay before starting next page
            setTimeout(() => {
              startSpeechSynthesis();
            }, 1000);
          });
        } else {
          stopSpeechSynthesis();
        }
      };
      
      // Start speaking
      speechSynthesis.speak(speechUtterance);
      isSpeaking = true;
    }
    
    function stopSpeechSynthesis() {
      if (speechSynthesis) {
        speechSynthesis.cancel();
      }
      
      isSpeaking = false;
      
      // Reset UI
      document.getElementById('read-aloud-btn').style.display = 'inline-flex';
      document.getElementById('stop-read-aloud-btn').style.display = 'none';
      
      // Remove highlight
      if (highlightedTextElement && highlightedTextElement.parentNode) {
        highlightedTextElement.parentNode.removeChild(highlightedTextElement);
        highlightedTextElement = null;
      }
    }
    
    function showNotification(message) {
      // Simple notification display
      const notification = document.createElement('div');
      notification.className = 'notification';
      notification.innerText = message;
      notification.style.position = 'fixed';
      notification.style.bottom = '20px';
      notification.style.right = '20px';
      notification.style.backgroundColor = '#4f46e5';
      notification.style.color = 'white';
      notification.style.padding = '12px 20px';
      notification.style.borderRadius = '4px';
      notification.style.boxShadow = '0 4px 6px rgba(0,0,0,0.1)';
      notification.style.zIndex = '9999';
      
      document.body.appendChild(notification);
      
      // Remove after 3 seconds
      setTimeout(() => {
        if (notification.parentNode) {
          notification.parentNode.removeChild(notification);
        }
      }, 3000);
    }
